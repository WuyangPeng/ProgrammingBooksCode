static char_type const *find(   char_type const *s,
                                size_type       cch,
                                char_type const &c)
{
    for(; 0 < cch; --cch, ++s)
    {
        if(eq(*s, c))
        {
            break;
        }
    }

    return (0 < cch) ? s : NULL;
}
